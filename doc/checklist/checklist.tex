\newcounter{testno}

% Tests
\section{Accuracy Tests}
\testrubric{\arabic{testno}}{8.1}{Command the vehicle to a setpoint of 3m, 5m, and 10m of its starting position in the x direction and ensure it ends within 10\% of the setpoint}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.1}{Command the vehicle to a setpoint of 3m, 5m, and 10m of its starting position in the y direction and ensure it ends within 10\% of the setpoint}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.1}{Command the vehicle to a setpoint of 3m, 5m, and 10m of its starting position in both the x and y directions and ensure it ends within 10\% of the setpoint}\\
\stepcounter{testno}

\section{Oscillation Tests}
\testrubric{\arabic{testno}}{8.2}{Command the vehicle to any setpoint and ensure it does not oscillate more than 0.05m in any x, y, or z direction}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.2}{Command the vehicle to any setpoint and ensure it does not oscillate more than 0.1 radians in the yaw direction}\\
\stepcounter{testno}

\section{State Estimation Tests}
\testrubric{\arabic{testno}}{8.3, 8.4}{Fly the vehicle manually to the heights of 0m, 0.5m and 1.5m. Read the log files and ensure that the estimated z is within +/-10\% at all of the heights.}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.3, 8.4}{Fly the vehicle manually in a 2m by 2m square three times. Read the log files and ensure that the estimated x and y are within +/-10\%.}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.3, 8.4}{Fly the vehicle manually in yaw from 0 to 2$\pi$ radians three times. Read the log files and ensure that the estimated yaw are within +/-10\%.}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.3, 8.4}{Fly the vehicle manually at a constant rate to cover a distance of 3m in 30 seconds in the x direction. Read the log files and ensure that the estimated x rate is within +/-10\%.}\\
\stepcounter{testno}

\testrubric{\arabic{testno}}{8.3, 8.4}{Fly the vehicle manually at a constant rate to cover a distance of 3m in 30 seconds in the y direction. Read the log files and ensure that the estimated y rate is within +/-10\%.}\\
\stepcounter{testno}
